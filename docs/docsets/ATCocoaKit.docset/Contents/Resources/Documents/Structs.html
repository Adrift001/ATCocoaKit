    UserDefaults Wrapper

    How to use

    // HandyJSON Type
class Person: HandyJSON {
    var name = ""
    required init() {}
}
@UserDefaultsWrapper(key: .handyJSON("person"), defaultValue: Person())
var person

// Base Type
@UserDefaultsWrapper(key: .base("username"), defaultValue: "")
var username: String
